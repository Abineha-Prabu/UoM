void disable_echo() {
    struct termios config;
    tcgetattr(STDIN_FILENO, &config);  
    config.c_lflag &= ~ECHO;           
    tcsetattr(STDIN_FILENO, TCSANOW, &config); 
}

void enable_echo() {
    struct termios config;
    tcgetattr(STDIN_FILENO, &config);   
    config.c_lflag |= ECHO;             
    tcsetattr(STDIN_FILENO, TCSANOW, &config); 
}
